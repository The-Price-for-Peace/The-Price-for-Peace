scripted_gui = {
	# Top bar button
    spaceflight_topbar_button = {
        context_type = player_context
        window_name = "spaceflight_topbar_button_container"
        
        visible = {
            always = yes
        }
        
        effects = {
            open_spaceflight_window_button_click = {
                if = {
                    limit = { has_country_flag = spaceflight_window_open }
                    clr_country_flag = spaceflight_window_open
                }
                else = {
                    set_country_flag = spaceflight_window_open
                }
            }
        }
	}
}
